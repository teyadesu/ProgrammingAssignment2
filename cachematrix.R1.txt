## This function will insert any value of the given variable "x" as the matrix function will set as cachematrix. The condition for this would be if not, x is NULL or x = NULL

makeCacheMatrix <- function(x = matrix()) {
  inv< - NULL
  set<-function(y) {
    x<<-y
    inv<< - NULL
  }

  get<-function() x
  setInv<-function(inverse) inv<<-inverse
  getInv<-function() inv
  list(set=set,
       get=get,
       setInv=setInv,
       getInv=getInv)
}

## This line will provide an inverse matrix wherein it caches the function so the used can retrieve it. The said cache will retrieve and send forward later on the function named as Solved_Cache


SolvedCache <- function(x, ...) {
  inv<-x$getInv()
  if(!is.null(inv)) {
    message("retrieving the matrix")
    return(inv)
  }

  ans<-x$get()
  inv<-solve(ans,...)
  x$setInv(inv)
  inv

}
